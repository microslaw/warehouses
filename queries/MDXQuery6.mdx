--Give an average grade from each course
SELECT
    [Dim Course].[Course ID].MEMBERS ON ROWS,
    [Measures].[Average_score_exam] ON COLUMNS
FROM [School DW];


--bez null
SELECT
    NON EMPTY ORDER([Dim Course].[Course ID].MEMBERS, [Measures].[Average_score_exam], BDESC) ON ROWS,
    [Measures].[Average_score_exam] ON COLUMNS
FROM [School DW];
