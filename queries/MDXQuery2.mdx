--Compare average survey scores between specializations.
SELECT
    [Dim Class].[Specialization].MEMBERS ON ROWS,
    [Measures].[Average_score_survey] ON COLUMNS
FROM
    [School DW];

-- bez pustych wartosci
SELECT
    NON EMPTY [Dim Class].[Specialization].MEMBERS ON ROWS,
    [Measures].[Average_score_survey] ON COLUMNS
FROM
    [School DW];
