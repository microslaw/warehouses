--Give the average exam grade of students from Stocznia.
SELECT
    [Dim Students].[Student ID].MEMBERS ON Rows,
    [Measures].[Average_score_exam] ON Columns
FROM
    [School Dw]
WHERE
    [Dim Students].[City District].[stocznia];

GO

-- bez nuli
SELECT
    NON EMPTY [Dim Students].[Student ID].MEMBERS ON ROWS,
    [Measures].[Average_score_exam] ON COLUMNS
FROM [School Dw]
WHERE [Dim Students].[City District].[stocznia];
